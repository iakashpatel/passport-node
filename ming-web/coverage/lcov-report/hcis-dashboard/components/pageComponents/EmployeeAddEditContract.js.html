import React from 'react';
import PropTypes from 'prop-types';
import DatePicker from 'react-datepicker';
import { Form, FormControl, FormGroup, Button, Col, ControlLabel } from 'react-bootstrap';
import AlertContainer from 'react-alert';
import moment from 'moment';
// import Router from 'next/router';
import { Link, Router } from '../../routes';
import Head from '../commonComponents/Head';
import Footer from '../commonComponents/Footer';
import Navbar from '../commonComponents/Navbar';
 
export class EmployeeAddEditContract extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      contractList: {
        start_date: this.props.isNewContract ? '' : moment(this.props.getContract.contract.start_date),
        end_date: this.props.isNewContract ? '' : moment(this.props.getContract.contract.end_date),
        type: this.props.isNewContract ? '' : this.props.getContract.contract.employee_type,
        salary: this.props.isNewContract ? '' : this.props.getContract.contract.salary,
      },
      alertOptions: {
        offset: 14,
        position: 'top right',
        theme: 'light',
        time: 0,
        transition: 'scale',
      },
    };
    this.onAlertClose = this.onAlertClose.bind(this);
    this.handleInputChange = this.handleInputChange.bind(this);
    this.handleChangeStart = this.handleChangeStart.bind(this);
    this.handleChangeEnd = this.handleChangeEnd.bind(this);
    this.handleSubmit = this.handleSubmit.bind(this);
    this.showAlert = this.showAlert.bind(this);
    this.showErrorAlert = this.showErrorAlert.bind(this);
  }
 
  onAlertClose() {
    const employeeId = this.props.url.query.employee_id;
    Router.push(`/employees/${employeeId}/contracts`);
  }
 
  handleSubmit(e) {
    e.preventDefault();
    const employeeId = this.props.url.query.employee_id;
    const editContractId = Number(this.props.url.query.contract_id);
    const addContract = {
      contract: {
        start_date: moment(this.state.contractList.start_date).format('YYYY-MM-DD'),
        end_date: moment(this.state.contractList.end_date).format('YYYY-MM-DD'),
        salary: this.state.contractList.salary,
        created_by: this.props.isNewContract ? this.props.admin.createdBy : null,
        updated_by: this.props.isNewContract ? null : this.props.admin.updatedBy,
        employee_type: this.state.contractList.type,
      },
      employee: {
        EmployeeBasic: {
          id: employeeId,
        },
      },
    };
    if (this.props.isNewContract) {
      if (addContract.contract.start_date && addContract.contract.end_date === 'Invalid date') {
        this.showErrorAlert();
      } else {
        this.props.actions.createSalaryDetails(addContract, employeeId);
        this.showAlert();
      }
    } else {
      this.props.actions.updateSalaryDetails(addContract, editContractId);
      this.showAlert();
    }
    Router.push(`/employees/${employeeId}/contracts`);
  }
 
  handleChangeStart(date) {
    const contractsData = this.state.contractList;
    contractsData.start_date = date;
    this.setState({ contractsData });
  }
 
  handleChangeEnd(date) {
    const contractsData = this.state.contractList;
    contractsData.end_date = date;
    this.setState({ contractsData });
  }
 
  handleInputChange(event) {
    const contractsData = this.state.contractList;
    contractsData[event.target.name] = event.target.name === 'salary' ? Number(event.target.value) : event.target.value;
    this.setState({ contractsData });
  }
 
  showAlert() {
    this.msg.show('SuccesFully Added', {
      time: 0,
      type: 'success',
    });
  }
 
  showErrorAlert() {
    this.msg.show('Invalid date', {
      // time: 0,
      type: 'error',
    });
  }
 
  render() {
    return (
      <div>
        <Head />
        <Navbar />
        <div className="container minheight">
          <div className="panel panel-default leaveListStyle">
            <div className="panel-heading salaryHeading">
              <span>{ this.props.isNewContract ? 'Add' : 'Edit' }&nbsp;</span>
              Employee Salary
              <div className="pull-right">
                <Link route="contracts" params={{ employee_id: this.props.url.query.employee_id }}>
                  <a className="btn btn-default">&lt;&nbsp;Back
                  </a>
                </Link>
              </div>
            </div>
            <div className="panel panel-body">
              <Form
                horizontal
                onSubmit={this.handleSubmit}
              >
                <br />
                <FormGroup controlId="formControlsSelect">
                  <Col componentClass={ControlLabel} sm={2}>
                    From Date<span className="star">*</span>
                  </Col>
                  <Col sm={3}>
                    <DatePicker
                      name="start_date"
                      selected={this.state.contractList.start_date}
                      onChange={this.handleChangeStart}
                      required
                      dateFormat="YYYY-MM-DD"
                      className="datepickerBox"
                    />
                  </Col>
                </FormGroup>
 
                <FormGroup controlId="formHorizontalDate">
                  <Col componentClass={ControlLabel} sm={2}>
                    To Date<span className="star">*</span>
                  </Col>
                  <Col sm={3}>
                    <DatePicker
                      name="end_date"
                      selected={this.state.contractList.end_date}
                      onChange={this.handleChangeEnd}
                      required
                      dateFormat="YYYY-MM-DD"
                      className="datepickerBox"
                    />
                  </Col>
                </FormGroup>
 
                <FormGroup controlId="formHorizontalPassword">
                  <Col componentClass={ControlLabel} sm={2}>
                    Salary Ammount<span className="star">*</span>
                  </Col>
                  <Col sm={3}>
                    <FormControl
                      name="salary"
                      type="number"
                      required
                      value={this.state.contractList.salary}
                      onChange={this.handleInputChange}
                    />
                  </Col>
                </FormGroup>
 
                <FormGroup controlId="formControlsTextarea">
                  <Col componentClass={ControlLabel} sm={2}>
                    Employee Type<span className="star">*</span>
                  </Col>
                  <Col sm={3}>
                    <FormControl
                      name="type"
                      componentClass="select"
                      value={this.state.contractList.type}
                      onChange={this.handleInputChange}
                    >
                      <option value="full_time">Full Time</option>
                      <option value="half_time">Half Time</option>
                      <option value="intern">Intern</option>
                    </FormControl>
                  </Col>
                </FormGroup>
 
                <br />
                <FormGroup>
                  <Col smOffset={2} sm={10}>
                    <div>
                      <AlertContainer ref={(a) => { this.msg = a; }} {...this.state.alertOptions} />
                      <Button
                        bsStyle="success"
                        type="submit"
                        value="Submit"
                      > Submit Details
                      </Button>
                    </div>
                  </Col>
                </FormGroup>
              </Form>
            </div>
          </div>
        </div>
        <Footer />
      </div>
    );
  }
}
 
EmployeeAddEditContract.propTypes = {
  isNewContract: PropTypes.bool,
  admin: PropTypes.object,
  url: PropTypes.object,
  employee_id: PropTypes.number,
  salary: PropTypes.number,
  employee_type: PropTypes.string,
  start_date:PropTypes.string,
  end_date: PropTypes.string,
  roles: PropTypes.oneOf(['readOnly', 'readWrite']),
  type: PropTypes.oneOf(['Half_time', 'Full_time', 'Intern']),
};
